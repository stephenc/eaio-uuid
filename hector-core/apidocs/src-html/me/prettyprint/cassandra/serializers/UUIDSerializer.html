001    package me.prettyprint.cassandra.serializers;
002    
003    import java.nio.ByteBuffer;
004    import java.util.UUID;
005    
006    /**
007     * A UUIDSerializer translates the byte[] to and from UUID types.
008     * @author Ed Anuff
009     *
010     */
011    public final class UUIDSerializer extends AbstractSerializer<UUID> {
012    
013      private static final UUIDSerializer instance = new UUIDSerializer();
014    
015      public static UUIDSerializer get() {
016        return instance;
017      }
018    
019      public ByteBuffer toByteBuffer(UUID uuid) {
020        if (uuid == null) {
021          return null;
022        }
023        long msb = uuid.getMostSignificantBits();
024        long lsb = uuid.getLeastSignificantBits();
025        byte[] buffer = new byte[16];
026    
027        for (int i = 0; i < 8; i++) {
028          buffer[i] = (byte) (msb >>> 8 * (7 - i));
029        }
030        for (int i = 8; i < 16; i++) {
031          buffer[i] = (byte) (lsb >>> 8 * (7 - i));
032        }
033    
034        return ByteBuffer.wrap(buffer);
035      }
036    
037      public UUID fromByteBuffer(ByteBuffer bytes) {
038        if (bytes == null) {
039          return null;
040        }    
041        return new UUID(bytes.getLong(), bytes.getLong());
042      }
043    
044    }
