001    package me.prettyprint.cassandra.serializers;
002    
003    import java.io.ByteArrayInputStream;
004    import java.io.ByteArrayOutputStream;
005    import java.io.IOException;
006    import java.io.ObjectInputStream;
007    import java.io.ObjectOutputStream;
008    import java.nio.ByteBuffer;
009    
010    import me.prettyprint.hector.api.Serializer;
011    import me.prettyprint.hector.api.exceptions.HectorSerializationException;
012    
013    /**
014     * The ObjectSerializer is used to turn objects into their binary representations.
015     *
016     * @author Bozhidar Bozhanov
017     *
018     */
019    public class ObjectSerializer extends AbstractSerializer<Object> implements Serializer<Object> {
020    
021      private static final ObjectSerializer INSTANCE = new ObjectSerializer();
022    
023      @Override
024      public ByteBuffer toByteBuffer(Object obj) {
025        try {
026          ByteArrayOutputStream baos = new ByteArrayOutputStream();
027          ObjectOutputStream oos = new ObjectOutputStream(baos);
028          oos.writeObject(obj);
029          oos.close();
030    
031          return ByteBuffer.wrap(baos.toByteArray());
032        } catch (IOException ex) {
033          throw new RuntimeException(ex);
034        }
035      }
036    
037      @Override
038      public Object fromByteBuffer(ByteBuffer bytes) {
039        if (bytes == null || !bytes.hasRemaining()) {
040          return null;
041        }
042        try {
043          ByteArrayInputStream bais = new ByteArrayInputStream(bytes.array());
044          ObjectInputStream ois = new ObjectInputStream(bais);
045          Object obj = ois.readObject();
046          ois.close();
047    
048          return obj;
049        } catch (Exception ex) {
050          throw new HectorSerializationException(ex);
051        }
052      }
053    
054      public static ObjectSerializer get() {
055        return INSTANCE;
056      }
057    
058    }
