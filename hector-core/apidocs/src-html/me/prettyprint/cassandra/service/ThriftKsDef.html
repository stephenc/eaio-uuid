001    package me.prettyprint.cassandra.service;
002    
003    import java.util.ArrayList;
004    import java.util.Collections;
005    import java.util.List;
006    import java.util.Map;
007    
008    import me.prettyprint.cassandra.utils.Assert;
009    import me.prettyprint.hector.api.ddl.ColumnFamilyDefinition;
010    import me.prettyprint.hector.api.ddl.KeyspaceDefinition;
011    
012    import org.apache.cassandra.thrift.KsDef;
013    import org.apache.commons.lang.builder.ToStringBuilder;
014    import org.apache.commons.lang.builder.ToStringStyle;
015    
016    public class ThriftKsDef implements KeyspaceDefinition {
017    
018      public static final String DEF_STRATEGY_CLASS = "org.apache.cassandra.locator.SimpleStrategy";
019      private final String name;
020      private String strategyClass;
021      private Map<String, String> strategyOptions;
022      private int replicationFactor;
023      private final List<ColumnFamilyDefinition> cfDefs;
024    
025      public ThriftKsDef(KsDef k) {
026        Assert.notNull(k, "KsDef is null");
027        name = k.name;
028        strategyClass = k.strategy_class;
029        strategyOptions = k.strategy_options;
030        replicationFactor = k.replication_factor;
031        cfDefs = ThriftCfDef.fromThriftList(k.cf_defs);
032      }
033    
034      public ThriftKsDef(String keyspaceName, String strategyClass, int replicationFactor,
035          List<ColumnFamilyDefinition> cfDefs) {
036        this.name = keyspaceName;
037        this.strategyClass = strategyClass;
038        this.replicationFactor = replicationFactor;
039        this.cfDefs = cfDefs;
040      }
041    
042      public ThriftKsDef(String keyspaceName) {
043        this.name = keyspaceName;
044        this.cfDefs = new ArrayList<ColumnFamilyDefinition>();
045        this.replicationFactor = 1;
046        this.strategyClass = DEF_STRATEGY_CLASS;
047      }
048      
049      public ThriftKsDef(KeyspaceDefinition keyspaceDefinition) {
050        name = keyspaceDefinition.getName();
051        strategyClass = keyspaceDefinition.getStrategyClass();
052        strategyOptions = keyspaceDefinition.getStrategyOptions();
053        replicationFactor = keyspaceDefinition.getReplicationFactor();
054        cfDefs = keyspaceDefinition.getCfDefs();
055      }
056    
057      public static List<KeyspaceDefinition> fromThriftList(List<KsDef> ks) {
058        if (ks == null || ks.isEmpty()) {
059          return Collections.emptyList();
060        }
061        List<KeyspaceDefinition> l = new ArrayList<KeyspaceDefinition>(ks.size());
062        for (KsDef k: ks) {
063          l.add(new ThriftKsDef(k));
064        }
065        return l;
066      }
067    
068      @Override
069      public String getName() {
070        return name;
071      }
072    
073      @Override
074      public String getStrategyClass() {
075        return strategyClass;
076      }
077    
078      @Override
079      public Map<String, String> getStrategyOptions() {
080        return Collections.unmodifiableMap(strategyOptions);
081      }
082    
083      @Override
084      public int getReplicationFactor() {
085        return replicationFactor;
086      }
087    
088      @Override
089      public List<ColumnFamilyDefinition> getCfDefs() {
090        return Collections.unmodifiableList(cfDefs);
091      }
092    
093      public KsDef toThrift() {
094        return new KsDef(name, strategyClass, replicationFactor, ThriftCfDef.toThriftList(cfDefs));
095      }
096    
097      public void setStrategyClass(String strategyClass) {
098        this.strategyClass = strategyClass;
099      }
100    
101      public void setStrategyOptions(Map<String, String> strategyOptions) {
102        this.strategyOptions = strategyOptions;
103      }
104    
105      public void setReplicationFactor(int replicationFactor) {
106        this.replicationFactor = replicationFactor;
107      }
108    
109      @Override
110      public String toString() {
111        return ToStringBuilder.reflectionToString(this, ToStringStyle.SHORT_PREFIX_STYLE);
112      }
113    
114    }
