083      /**
084       *
085       * @param clusterName The cluster name. This is an identifying string for the cluster, e.g.
086       * "production" or "test" etc. Clusters will be created on demand per each unique clusterName key.
087       * @param hostIp host:ip format string
088       * @return
089       */
090      public static Cluster getOrCreateCluster(String clusterName, String hostIp) {    
091        return getOrCreateCluster(clusterName, new CassandraHostConfigurator(hostIp));
092      }
093    
094      public static Cluster getOrCreateCluster(String clusterName,
095          CassandraHostConfigurator cassandraHostConfigurator) {
096        synchronized (clusters) {
097          Cluster c = clusters.get(clusterName);
098          if (c == null) {
099            c = createCluster(clusterName, cassandraHostConfigurator);
100            clusters.put(clusterName, c);
101          }
102          return c;
103        }
104      }
105    
106      public static Cluster createCluster(String clusterName, CassandraHostConfigurator cassandraHostConfigurator) {    
107        return clusters.get(clusterName) == null ? new ThriftCluster(clusterName, cassandraHostConfigurator) : clusters.get(clusterName);
108      }

315      /**
316       * Create a clock resolution based on <code>clockResolutionName</code> which has to match any of the constants defined
317       * at {@link ClockResolution}
318       * @param clockResolutionName type of clock resolution to create
319       * @return a ClockResolution
320       */
321      public static ClockResolution createClockResolution(String clockResolutionName) {
322          if (clockResolutionName.equals(ClockResolution.SECONDS)) {
323              return new SecondsClockResolution();
324          } else if (clockResolutionName.equals(ClockResolution.MILLISECONDS)) {
325              return new MillisecondsClockResolution();
326          } else if (clockResolutionName.equals(ClockResolution.MICROSECONDS)) {
327              return new MicrosecondsClockResolution();
328          } else if (clockResolutionName.equals(ClockResolution.MICROSECONDS_SYNC)) {
329              return new MicrosecondsSyncClockResolution();
330          }
331          throw new RuntimeException(String.format("Unsupported clock resolution: %s", clockResolutionName));
332      }
