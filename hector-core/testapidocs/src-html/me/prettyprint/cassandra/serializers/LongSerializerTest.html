001    package me.prettyprint.cassandra.serializers;
002    
003    import static org.junit.Assert.assertEquals;
004    
005    import me.prettyprint.cassandra.serializers.LongSerializer;
006    
007    import org.junit.Test;
008    
009    public class LongSerializerTest {
010    
011      @Test
012      public void testConversions() {
013        test(0l);
014        test(1l);
015        test(-1l);
016        test(Long.MAX_VALUE);
017        test(Long.MIN_VALUE);
018        test(null);
019      }
020    
021      private void test(Long number) {
022        LongSerializer ext = new LongSerializer();
023        assertEquals(number, ext.fromByteBuffer(ext.toByteBuffer(number)));
024      }
025    }
