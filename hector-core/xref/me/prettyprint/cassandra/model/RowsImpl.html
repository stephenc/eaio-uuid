17   * Returned by a MultigetSliceQuery (multiget_slice thrift call)
18   *
19   * @author Ran Tavory
20   *
21   * @param <N>
22   * @param <V>
23   */
24  public class RowsImpl<K, N, V> implements Rows<K, N, V> {
25  
26    protected final Map<K, Row<K, N, V>> rows;
27  
28    public RowsImpl(Map<K, List<Column>> thriftRet, Serializer<N> nameSerializer,
29        Serializer<V> valueSerializer) {
30      Assert.noneNull(thriftRet, nameSerializer, valueSerializer);
31      rows = new LinkedHashMap<K, Row<K, N, V>>(thriftRet.size());
32      for (Map.Entry<K, List<Column>> entry : thriftRet.entrySet()) {
33        rows.put(entry.getKey(), new RowImpl<K, N, V>(entry.getKey(), entry.getValue(), nameSerializer,
34            valueSerializer));
35      }
36    }
37  
38    @Override
39    public Row<K, N, V> getByKey(K key) {
40      return rows.get(key);
41    }
42  
43    @Override
44    public int getCount() {
45      return rows.size();
46    }
47  
48    @Override
49    public Iterator<Row<K, N, V>> iterator() {
50      return rows.values().iterator();
51    }
52  
53    @Override
54    public String toString() {
55      return "Rows(" + rows + ")";
56    }
