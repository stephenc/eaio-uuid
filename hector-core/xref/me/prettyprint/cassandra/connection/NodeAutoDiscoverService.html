25    public NodeAutoDiscoverService(HConnectionManager connectionManager,
26        CassandraHostConfigurator cassandraHostConfigurator) {
27      super(connectionManager, cassandraHostConfigurator);
28      this.retryDelayInSeconds = cassandraHostConfigurator.getAutoDiscoveryDelayInSeconds();
29      sf = executor.scheduleWithFixedDelay(new QueryRing(), retryDelayInSeconds,retryDelayInSeconds, TimeUnit.SECONDS);
30    }
31  
32    @Override
33    void shutdown() {
34      log.error("Auto Discovery retry shutdown hook called");
35      if ( sf != null ) {
36        sf.cancel(true);
37      }
38      if ( executor != null ) {
39        executor.shutdownNow();
40      }
41      log.error("AutoDiscovery retry shutdown complete");
42    }

44    @Override
45    public void applyRetryDelay() {
46      // no op for now
47    }

49    class QueryRing implements Runnable {
50  
51      @Override
52      public void run() {
53        if ( log.isDebugEnabled() ) {
54          log.debug("Auto discovery service running...");
55        }
56        Set<CassandraHost> foundHosts = discoverNodes();
57        if ( foundHosts != null && foundHosts.size() > 0 ) {
58          log.info("Found {} new host(s) in Ring", foundHosts.size());
59          for (CassandraHost cassandraHost : foundHosts) {
60            log.info("Addding found host {} to pool", cassandraHost);
61            cassandraHostConfigurator.applyConfig(cassandraHost);
62            connectionManager.addCassandraHost(cassandraHost);
63          }
64        }
65        if ( log.isDebugEnabled() ) {
66          log.debug("Auto discovery service run complete.");
67        }
68      }
69  
70    }

72    public Set<CassandraHost> discoverNodes() {
73      Set<CassandraHost> existingHosts = connectionManager.getHosts();
74      Set<CassandraHost> foundHosts = new HashSet<CassandraHost>();
75  
76      HThriftClient thriftClient = null;
77      log.info("using existing hosts {}", existingHosts);
78      try {
79        thriftClient = connectionManager.borrowClient();
80  
81        for (KsDef keyspace : thriftClient.getCassandra().describe_keyspaces()) {
82          if (!keyspace.getName().equals(Keyspace.KEYSPACE_SYSTEM)) {
83            List<TokenRange> tokenRanges = thriftClient.getCassandra().describe_ring(keyspace.getName());
84            for (TokenRange tokenRange : tokenRanges) {
85              for (String host : tokenRange.getEndpoints()) {
86                CassandraHost foundHost = new CassandraHost(host,cassandraHostConfigurator.getPort());
87                if ( !existingHosts.contains(foundHost) ) {
88                  log.info("Found a node we don't know about {} for TokenRange {}", foundHost, tokenRange);
89                  foundHosts.add(foundHost);
90                }
91              }
92            }
93            break;
94          }
95        }
96      } catch (Exception e) {
97        log.error("Downed Host retry failed attempt to verify CassandraHost", e);
98      } finally {
99        connectionManager.releaseClient(thriftClient);
100     }
101     return foundHosts;
102   }
