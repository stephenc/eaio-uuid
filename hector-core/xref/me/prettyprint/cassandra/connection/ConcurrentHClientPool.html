32    public ConcurrentHClientPool(CassandraHost host) {
33      this.cassandraHost = host;
34  
35      availableClientQueue = new ArrayBlockingQueue<HThriftClient>(cassandraHost.getMaxActive(), true);
36      activeClients = new NonBlockingHashSet<HThriftClient>();
37      numActive = new AtomicInteger();
38      numBlocked = new AtomicInteger();
39      active = new AtomicBoolean(true);
40  
41      maxWaitTimeWhenExhausted = cassandraHost.getMaxWaitTimeWhenExhausted() < 0 ? 0 : cassandraHost.getMaxWaitTimeWhenExhausted();
42  
43      for (int i = 0; i < cassandraHost.getMaxActive()/3; i++) {
44        availableClientQueue.add(new HThriftClient(cassandraHost).open());
45      }
46      if ( log.isDebugEnabled() ) {
47        log.debug("Concurrent Host pool started with {} active clients; max: {} exhausted wait: {}",
48            new Object[]{getNumIdle(),
49            cassandraHost.getMaxActive(),
50            maxWaitTimeWhenExhausted});
51      }

55    public HThriftClient borrowClient() throws HectorException {
56      if ( !active.get() ) {
57        throw new HectorException("Attempt to borrow on in-active pool: " + getName());
58      }
59      HThriftClient cassandraClient;
60      int currentActive = numActive.incrementAndGet();
61      int tillExhausted = cassandraHost.getMaxActive() - currentActive;
62  
63      numBlocked.incrementAndGet();
64      cassandraClient = availableClientQueue.poll();
65      if ( cassandraClient == null ) {
66        if ( tillExhausted > 0 ) {
67          // if we start with #of threads == getMaxActive, we could trigger this condition
68          addClientToPoolGently(new HThriftClient(cassandraHost).open());
69          log.debug("created new client. NumActive:{} untilExhausted: {}", currentActive, tillExhausted);
70        }
71        // blocked take on the queue if we are configured to wait forever
72        if ( log.isDebugEnabled() ) {
73          log.debug("blocking on queue - current block count {}", numBlocked.get());
74        }
75        // wait and catch, creating a new one if the counts have changed. Infinite wait should just recurse.
76        if ( maxWaitTimeWhenExhausted == 0 ) {
77          while (cassandraClient == null && active.get() ) {
78            try {
79              cassandraClient = availableClientQueue.poll(100, TimeUnit.MILLISECONDS);
80            } catch (InterruptedException ie) {
81              log.error("InterruptedException poll operation on retry forever", ie);
82              break;
83            }
84          }
85        } else {
86  
87          try {
88            cassandraClient = availableClientQueue.poll(maxWaitTimeWhenExhausted, TimeUnit.MILLISECONDS);
89            if ( cassandraClient == null ) {
90              numBlocked.decrementAndGet();
91              throw new PoolExhaustedException(String.format("maxWaitTimeWhenExhausted exceeded for thread %s on host %s",
92                  new Object[]{
93                  Thread.currentThread().getName(),
94                  cassandraHost.getName()}
95              ));
96            }
97          } catch (InterruptedException ie) {
98            //monitor.incCounter(Counter.POOL_EXHAUSTED);
99            numActive.decrementAndGet();
100         }
101       }
102 
103     }
104     if ( cassandraClient == null ) {
105       throw new HectorException("HConnectionManager returned a null client after aquisition - are we shutting down?");
106     }
107     activeClients.add(cassandraClient);
108     numBlocked.decrementAndGet();
109 
110 
111     return cassandraClient;
112   }

114   /**
115    * Controlled shutdown of pool. Go through the list of available clients
116    * in the queue and call {@link HThriftClient#close()} on each. Toggles
117    * a flag to indicate we are going into shutdown mode. Any subsequent calls
118    * will throw an IllegalArgumentException.
119    *
120    *
121    */
122   void shutdown() {
123     if (!active.compareAndSet(true, false) ) {
124       throw new IllegalArgumentException("shutdown() called for inactive pool: " + getName());
125     }
126     log.error("Shutdown triggered on {}", getName());
127     Set<HThriftClient> clients = new HashSet<HThriftClient>();
128     availableClientQueue.drainTo(clients);
129     if ( clients.size() > 0 ) {
130       for (HThriftClient hThriftClient : clients) {
131         hThriftClient.close();
132       }
133     }
134     log.error("Shutdown complete on {}", getName());
135   }

187   public void releaseClient(HThriftClient client) throws HectorException {
188     activeClients.remove(client);
189     numActive.decrementAndGet();
190     boolean open = client.isOpen();
191     if ( open ) {
192       if ( active.get() ) {
193         addClientToPoolGently(client);
194       } else {
195         log.info("Open client {} released to in-active pool for host {}. Closing.", client, cassandraHost);
196         client.close();
197       }
198     } else {
199       if ( activeClients.size() < getMaxActive() && numBlocked.get() > 0) {
200         addClientToPoolGently(new HThriftClient(cassandraHost).open());
201       }
202     }
203 
204     if ( log.isDebugEnabled() ) {
205       log.debug("Status of releaseClient {} to queue: {}", client.toString(), open);
206     }
207   }

209   /**
210    * Avoids a race condition on adding clients back to the pool if pool is almost full.
211    * Almost always a result of batch operation startup and shutdown (when multiple threads
212    * are releasing at the same time).
213    * @param client
214    */
215   private void addClientToPoolGently(HThriftClient client) {
216     try {
217       availableClientQueue.add(client);
218     } catch (IllegalStateException ise) {
219       log.error("Capacity hit adding client back to queue. Closing extra.");
220       client.close();
221     }
222   }
