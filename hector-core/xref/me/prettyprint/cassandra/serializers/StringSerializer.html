11  public final class StringSerializer extends AbstractSerializer<String> {

13    private static final String UTF_8 = "UTF-8";
14    private static final StringSerializer instance = new StringSerializer();
15  
16    public static StringSerializer get() {
17      return instance;
18    }
19  
20    @Override
21    public ByteBuffer toByteBuffer(String obj) {
22      if (obj == null) {
23        return null;
24      }
25      try {
26        return ByteBuffer.wrap(obj.getBytes(UTF_8));
27      } catch (UnsupportedEncodingException e) {
28        throw new RuntimeException(e);
29      }
30    }
31  
32    @Override
33    public String fromByteBuffer(ByteBuffer byteBuffer) {
34      if (byteBuffer == null) {
35        return null;
36      }
37      try {
38      return new String(byteBuffer.array(), byteBuffer.arrayOffset() + byteBuffer.position(),
39          byteBuffer.remaining(), UTF_8);
40      } catch (UnsupportedEncodingException e) {
41        // not really possible
42        e.printStackTrace();
43      }
44      return null;
45    }
