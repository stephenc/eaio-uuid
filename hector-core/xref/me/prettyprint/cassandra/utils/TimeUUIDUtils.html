53    private static long createTime(long currentTime) {
54      long time;
55  
56      // UTC time
57      long timeToUse = (currentTime * 10000) + 0x01B21DD213814000L;
58  
59      // time low
60      time = timeToUse << 32;
61  
62      // time mid
63      time |= (timeToUse & 0xFFFF00000000L) >> 16;
64  
65      // time hi and version
66      time |= 0x1000 | ((timeToUse >> 48) & 0x0FFF); // version 1
67      return time;
68    }

92    /**
93     * As byte array.
94     * This method is often used in conjunction with @link {@link #getTimeUUID()}
95     *
96     * @param uuid the uuid
97     *
98     * @return the byte[]
99     */
100   public static byte[] asByteArray(java.util.UUID uuid) {
101     long msb = uuid.getMostSignificantBits();
102     long lsb = uuid.getLeastSignificantBits();
103     byte[] buffer = new byte[16];
104 
105     for (int i = 0; i < 8; i++) {
106       buffer[i] = (byte) (msb >>> 8 * (7 - i));
107     }
108 
109     for (int i = 8; i < 16; i++) {
110       buffer[i] = (byte) (lsb >>> 8 * (7 - i));
111     }
112 
113     return buffer;
114   }
