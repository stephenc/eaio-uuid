11  public class MicrosecondsSyncClockResolution extends AbstractClockResolution implements ClockResolution {

13    private static final long ONE_THOUSAND = 1000L;

15      /**
16       * The last time value issued. Used to try to prevent duplicates.
17       */
18    private static long lastTime = -1;
19  
20    @Override
21    public long createClock() {
22      // The following simulates a microseconds resolution by advancing a static counter
23      // every time a client calls the createClock method, simulating a tick.
24      long us = getSystemMilliseconds() * ONE_THOUSAND;
25      // Synchronized to guarantee unique time within and across threads.
26      synchronized (MicrosecondsSyncClockResolution.class) {
27        if (us > lastTime) {
28          lastTime = us;
29        } else {
30          // the time i got from the system is equals or less
31          // (hope not - clock going backwards)
32          // One more "microsecond"
33          us = ++lastTime;
34        }
35      }
36      return us;
37    }
