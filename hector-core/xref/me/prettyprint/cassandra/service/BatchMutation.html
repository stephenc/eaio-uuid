18  /**
19   * A BatchMutation object is used to construct the {@link KeyspaceService#batchMutate(BatchMutation)} call.
20   *
21   * A BatchMutation encapsulates a set of updates (or insertions) and deletions all submitted at the
22   * same time to cassandra. The BatchMutation object is useful for user friendly construction of
23   * the thrift call batch_mutate.
24   *
25   * @author Ran Tavory (rantan@outbrain.com)
26   * @author Nathan McCall (nate@riptano.com)
27   *
28   */
29  public final class BatchMutation<K> {
30  
31    private final Map<ByteBuffer,Map<String,List<Mutation>>> mutationMap;
32    private final Serializer<K> keySerializer;
33  
34    public BatchMutation(Serializer<K> serializer) {
35      this.keySerializer = serializer;
36      mutationMap = new HashMap<ByteBuffer,Map<String,List<Mutation>>>();
37    }
38  
39    private BatchMutation(Serializer<K> serializer, Map<ByteBuffer,Map<String,List<Mutation>>> mutationMap) {
40      this.keySerializer = serializer;
41      this.mutationMap = mutationMap;
42    }
43  
44    /**
45     * Add an Column insertion (or update) to the batch mutation request.
46     */
47    public BatchMutation<K> addInsertion(K key, List<String> columnFamilies,
48        Column column) {
49      Mutation mutation = new Mutation();
50      mutation.setColumn_or_supercolumn(new ColumnOrSuperColumn().setColumn(column));
51      addMutation(key, columnFamilies, mutation);
52      return this;
53    }
54  
55  
56    /**
57     * Add an SuperColumn insertion (or update) to the batch mutation request.
58     */
59    public BatchMutation<K> addSuperInsertion(K key, List<String> columnFamilies,
60        SuperColumn superColumn) {
61      Mutation mutation = new Mutation();
62      mutation.setColumn_or_supercolumn(new ColumnOrSuperColumn().setSuper_column(superColumn));
63      addMutation(key, columnFamilies, mutation);
64      return this;
65    }
66  
67    /**
68     * Add a deletion request to the batch mutation.
69     */
70    public BatchMutation<K> addDeletion(K key, List<String> columnFamilies, Deletion deletion) {
71      Mutation mutation = new Mutation();
72      mutation.setDeletion(deletion);
73      addMutation(key, columnFamilies, mutation);
74      return this;
75    }
76  
77    private void addMutation(K key, List<String> columnFamilies, Mutation mutation) {
78      Map<String, List<Mutation>> innerMutationMap = getInnerMutationMap(key);
79      for (String columnFamily : columnFamilies) {
80        if (innerMutationMap.get(columnFamily) == null) {
81          innerMutationMap.put(columnFamily, Arrays.asList(mutation));
82        } else {
83          List<Mutation> mutations = new ArrayList<Mutation>(innerMutationMap.get(columnFamily));
84          mutations.add(mutation);
85          innerMutationMap.put(columnFamily, mutations);
86        }
87      }
88      mutationMap.put(keySerializer.toByteBuffer(key), innerMutationMap);
89    }
90  
91    private Map<String, List<Mutation>> getInnerMutationMap(K key) {
92      Map<String, List<Mutation>> innerMutationMap = mutationMap.get(keySerializer.toByteBuffer(key));
93      if (innerMutationMap == null) {
94        innerMutationMap = new HashMap<String, List<Mutation>>();
95      }
96      return innerMutationMap;
97    }
98  
99    Map<ByteBuffer,Map<String,List<Mutation>>> getMutationMap() {
100     return mutationMap;
101   }
