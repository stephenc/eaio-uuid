19  /**
20   * A cluster instance the client side representation of a cassandra server cluster.
21   *
22   * The cluster is usually the main entry point for programs using hector. To start operating on
23   * cassandra cluster you first get or create a cluster, then a keyspace operator for the keyspace
24   * you're interested in and then create mutations of queries
25   * <code>
26   * //get a cluster:
27   * Cluster cluster = getOrCreateCluster("MyCluster", new CassandraHostConfigurator("127.0.0.1:9170"));
28   * //get a keyspace from this cluster:
29   * Keyspace ko = createKeyspace("Keyspace1", cluster);
30   * //Create a mutator:
31   * Mutator m = createMutator(ko);
32   * // Make a mutation:
33   * MutationResult mr = m.insert("key", cf, createColumn("name", "value", serializer, serializer));
34   * </code>
35   *
36   * THREAD SAFETY: This class is thread safe.
37   *
38   * @author Ran Tavory
39   * @author zznate
40   */
41  public abstract class AbstractCluster implements Cluster {

61    public AbstractCluster(String clusterName, CassandraHostConfigurator cassandraHostConfigurator, Map<String, String> credentials) {
62      connectionManager = new HConnectionManager(cassandraHostConfigurator);
63      name = clusterName;
64      configurator = cassandraHostConfigurator;
65      failoverPolicy = FailoverPolicy.ON_FAIL_TRY_ALL_AVAILABLE;
66      cassandraClientMonitor = JmxMonitor.getInstance(connectionManager).getCassandraMonitor();
67      xtrans = new ExceptionsTranslatorImpl();
68      clockResolution = cassandraHostConfigurator.getClockResolution();
69      this.credentials = Collections.unmodifiableMap(credentials);
70    }

121     Operation<List<KeyspaceDefinition>> op = new Operation<List<KeyspaceDefinition>>(OperationType.META_READ, getCredentials()) {
122       @Override
123       public List<KeyspaceDefinition> execute(Cassandra.Client cassandra) throws HectorException {
124         try {
125           return ThriftKsDef.fromThriftList(cassandra.describe_keyspaces());
126         } catch (Exception e) {
127           throw xtrans.translate(e);

129       }
130     };

178   throws HectorException {
179     Operation<KeyspaceDefinition> op = new Operation<KeyspaceDefinition>(
180         OperationType.META_READ, getCredentials()) {
181       @Override
182       public KeyspaceDefinition execute(Cassandra.Client cassandra)
183       throws HectorException {
184         try {
185           return new ThriftKsDef(cassandra.describe_keyspace(keyspace));
186         } catch (org.apache.cassandra.thrift.NotFoundException nfe) {
187           setException(xtrans.translate(nfe));
188           return null;
189         } catch (Exception e) {
190           throw xtrans.translate(e);
191         }
192       }
193     };
194     connectionManager.operateWithFailover(op);
195     return op.getResult();
