94    public static Cluster getOrCreateCluster(String clusterName,
95        CassandraHostConfigurator cassandraHostConfigurator) {
96      synchronized (clusters) {
97        Cluster c = clusters.get(clusterName);
98        if (c == null) {
99          c = createCluster(clusterName, cassandraHostConfigurator);
100         clusters.put(clusterName, c);
101       }
102       return c;
103     }
104   }

315   /**
316    * Create a clock resolution based on <code>clockResolutionName</code> which has to match any of the constants defined
317    * at {@link ClockResolution}
318    * @param clockResolutionName type of clock resolution to create
319    * @return a ClockResolution
320    */
321   public static ClockResolution createClockResolution(String clockResolutionName) {
322       if (clockResolutionName.equals(ClockResolution.SECONDS)) {
323           return new SecondsClockResolution();
324       } else if (clockResolutionName.equals(ClockResolution.MILLISECONDS)) {
325           return new MillisecondsClockResolution();
326       } else if (clockResolutionName.equals(ClockResolution.MICROSECONDS)) {
327           return new MicrosecondsClockResolution();
328       } else if (clockResolutionName.equals(ClockResolution.MICROSECONDS_SYNC)) {
329           return new MicrosecondsSyncClockResolution();
330       }
331       throw new RuntimeException(String.format("Unsupported clock resolution: %s", clockResolutionName));
332   }
