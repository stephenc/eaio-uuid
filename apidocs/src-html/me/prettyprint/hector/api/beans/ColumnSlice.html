001    package me.prettyprint.hector.api.beans;
002    
003    import java.util.List;
004    
005    
006    /**
007     * A ColumnSlice represents a set of columns as returned by calls such as get_slice
008     *
009     * @author ran
010     */
011    public interface ColumnSlice<N, V> {
012    
013      /**
014       *
015       * @return an unmodifiable list of the columns
016       */
017      List<HColumn<N, V>> getColumns();
018    
019      HColumn<N, V> getColumnByName(N columnName);
020    
021    }
