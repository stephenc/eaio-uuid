001    package me.prettyprint.hector.api.exceptions;
002    
003    /**
004     * Base exception class for all Hector related exceptions.
005     * 
006     * @author Ran Tavory (rantav@gmail.com)
007     *
008     */
009    public class HectorException extends RuntimeException {
010    
011      private static final long serialVersionUID = -8498691501268563571L;
012    
013      public HectorException(String msg) {
014        super(msg);
015      }
016      
017      public HectorException(Throwable t) {
018        super(t);
019      }
020    
021      public HectorException(String s, Throwable t) {
022        super(s, t);
023      }
024    }
