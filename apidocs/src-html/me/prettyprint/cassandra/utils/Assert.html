001    package me.prettyprint.cassandra.utils;
002    
003    /**
004     * A generic low weight assert utility, very similar with Spring's Assert class,
005     * just without the dependency on Spring
006     *
007     * See for example
008     * http://www.jarvana.com/jarvana/view/org/springframework/spring
009     * /1.2.9/spring-1.2.9-javadoc.jar!/org/springframework/util/Assert.html
010     *
011     * @author Ran Tavory
012     *
013     */
014    public final class Assert {
015    
016      public static void notNull(Object object, String message) {
017        if (object == null) {
018          throw new IllegalArgumentException(message);
019        }
020      }
021    
022      public static void noneNull(Object... object) {
023        for (int i = 0; i < object.length; ++i) {
024          if (object[i] == null) {
025            throw new NullPointerException("Null not allowed, number " + (i + 1));
026          }
027        }
028      }
029    
030      public static void isTrue(boolean b, String message) {
031        if (!b) {
032          throw new IllegalArgumentException(message);
033        }
034      }
035    }
