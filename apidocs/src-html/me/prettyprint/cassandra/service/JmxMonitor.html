003    import java.lang.management.ManagementFactory;
004    import java.lang.reflect.InvocationTargetException;
005    import java.lang.reflect.Method;
006    import java.net.URL;
007    
008    import javax.management.InstanceAlreadyExistsException;
009    import javax.management.MBeanRegistrationException;
010    import javax.management.MBeanServer;
011    import javax.management.MalformedObjectNameException;
012    import javax.management.NotCompliantMBeanException;
013    import javax.management.ObjectName;
014    
015    import me.prettyprint.cassandra.connection.HConnectionManager;
016    
017    import org.slf4j.Logger;
018    import org.slf4j.LoggerFactory;

020    /**
021     * A JMX monitor singlton.
022     *
023     * @author Ran Tavory (ran@outbain.com)
024     *
025     */
026    public class JmxMonitor {
027    
028      private final Logger log = LoggerFactory.getLogger(getClass());
029    
030      private MBeanServer mbs;
031      private CassandraClientMonitor cassandraClientMonitor;
032      private static JmxMonitor monitorInstance;
033    
034      private JmxMonitor(HConnectionManager connectionManager) {
035        CassandraClientMonitor cassandraClientMonitor = new CassandraClientMonitor(connectionManager);
036        mbs = ManagementFactory.getPlatformMBeanServer();
037        this.cassandraClientMonitor = cassandraClientMonitor;
038        try {
039          registerMonitor("me.prettyprint.cassandra.service", "hector",
040              cassandraClientMonitor);
041        } catch (MalformedObjectNameException e) {
042          log.error("Unable to register JMX monitor", e);
043        } catch (InstanceAlreadyExistsException e) {
044          log.error("Unable to register JMX monitor", e);
045        } catch (MBeanRegistrationException e) {
046          log.error("Unable to register JMX monitor", e);
047        } catch (NotCompliantMBeanException e) {
048          log.error("Unable to register JMX monitor", e);
049        }
050      }
051    
052      public static JmxMonitor getInstance(HConnectionManager connectionManager) {
053        if ( monitorInstance == null ) {
054          monitorInstance = new JmxMonitor(connectionManager);
055        }
056        return monitorInstance;
057      }
058    
059    
060      public void registerMonitor(String name, String monitorType, Object monitoringInterface)
061          throws MalformedObjectNameException, InstanceAlreadyExistsException,
062          MBeanRegistrationException, NotCompliantMBeanException {
063    
064        String monitorName = generateMonitorName(name, monitorType);
065        log.info("Registering JMX {}", monitorName);
066    
067        ObjectName oName = new ObjectName(monitorName);
068    
069        // Check if the monitor is already registered
070        if (mbs.isRegistered(oName)) {
071          log.info("Monitor already registered: {}", oName);
072          return;
073        }
074    
075        mbs.registerMBean(monitoringInterface, oName);
076    
077        // Register perf4j monitors
078        if ("true".equalsIgnoreCase(
079            System.getProperty("com.prettyprint.cassandra.load_hector_log4j", "true"))) {
080          registerPerf4J();
081        }
082      }
083    
084      private void registerPerf4J() {
085        URL url = getClass().getClassLoader().getResource("hectorLog4j.xml");
086        if (url == null) {
087          log.warn("Unable to locate hectorLog4j.xml; performance counters will not be exported");
088        } else {
089          try {
090            final Class<?> domConfiguratorClass = getClass().getClassLoader().loadClass("org.apache.log4j.xml.DOMConfigurator");
091            final Method method = domConfiguratorClass.getMethod( "configure", URL.class );
092            method.invoke( null, url );
093          } catch( ClassNotFoundException e ) {
094            log.warn("Unable to load log4j's DOMConfigurator. Performance counters will not be exported. To fix, include the log4j jar in your application's classpath.");
095          } catch( SecurityException e ) {
096            log.error( "Could not access method DOMConfigurator.configure(URL)", e );
097          } catch( NoSuchMethodException e ) {
098            log.error( "Could not find method DOMConfigurator.configure(URL)", e );
099          } catch( IllegalArgumentException e ) {
100            log.error( "Could not invoke method DOMConfigurator.configure(URL)", e );
101          } catch( IllegalAccessException e ) {
102            log.error( "Could not invoke method DOMConfigurator.configure(URL)", e );
103          } catch( InvocationTargetException e ) {
104            throw new RuntimeException(e.getCause());
105          }
106        }
107      }
108    
109      private String generateMonitorName(String className, String monitorType) {
110        StringBuilder sb = new StringBuilder();
111        sb.append(className);
112        sb.append(":ServiceType=");
113        // append the classloader name so we have unique names in web apps.
114        sb.append(getUniqueClassloaderIdentifier());
115        if (null != monitorType && monitorType.length() > 0) {
116          sb.append(",MonitorType=" + monitorType);
117        }
118        return sb.toString();
119      }
120    
121      /**
122       * Generates a unique, but still nice and predictable name representing this classloader so that
123       * even apps operating under a web server such as tomcat with multiple classloaders would bee able
124       * to register each with its own unique mbean.
125       */
126      private String getUniqueClassloaderIdentifier() {
127        String contextPath = getContextPath();
128        if (contextPath != null) {
129          return contextPath;
130        }
131        return "hector";
132      }
133    
134      /**
135       * Tries to guess a context path for the running application.
136       * If this is a web application running under a tomcat server this will work.
137       * If unsuccessful, returns null.
138       * @return A string representing the current context path or null if it cannot be determined.
139       */
140      private String getContextPath() {
141        URL url = getClass().getClassLoader().getResource("/");
142        if (url != null) {
143          String[] elements = url.toString().split("/");
144          for (int i = elements.length - 1; i > 0; --i) {
145            // URLs look like this: file:/.../ImageServer/WEB-INF/classes/
146            // And we want that part that's just before WEB-INF
147            if ("WEB-INF".equals(elements[i])) {
148              return elements[i - 1];
149            }
150          }
151        }
152        return null;
153      }
154    
155      public CassandraClientMonitor getCassandraMonitor() {
156        return cassandraClientMonitor;
157      }
