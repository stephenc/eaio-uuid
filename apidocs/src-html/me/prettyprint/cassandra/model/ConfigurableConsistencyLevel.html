001    package me.prettyprint.cassandra.model;
002    
003    import java.util.HashMap;
004    import java.util.Map;
005    
006    import me.prettyprint.cassandra.service.OperationType;
007    import me.prettyprint.hector.api.HConsistencyLevel;
008    import me.prettyprint.hector.api.ConsistencyLevelPolicy;
009    
010    import org.slf4j.Logger;
011    import org.slf4j.LoggerFactory;
012    
013    /**
014     * Configurable and Runtime adjustable ConsistencyLevelPolicy
015     * @author zznate
016     */
017    public class ConfigurableConsistencyLevel implements ConsistencyLevelPolicy {
018      private final Logger log = LoggerFactory.getLogger(ConfigurableConsistencyLevel.class);
019    
020      private Map<String, HConsistencyLevel> readCfConsistencyLevels = new HashMap<String, HConsistencyLevel>();
021      private Map<String, HConsistencyLevel> writeCfConsistencyLevels = new HashMap<String, HConsistencyLevel>();
022      private HConsistencyLevel defaultReadConsistencyLevel = HConsistencyLevel.QUORUM;
023      private HConsistencyLevel defaultWriteConsistencyLevel = HConsistencyLevel.QUORUM;
024    
025      @Override
026      public HConsistencyLevel get(OperationType op) {
027        return op.equals(OperationType.READ) ? defaultReadConsistencyLevel : defaultWriteConsistencyLevel;
028      }
029    
030      @Override
031      public HConsistencyLevel get(OperationType op, String cfName) {
032        if (op.equals(OperationType.READ)) {
033          HConsistencyLevel rcf = readCfConsistencyLevels.get(cfName);
034          return rcf != null ? rcf : defaultReadConsistencyLevel;
035        } else {
036          HConsistencyLevel wcf = writeCfConsistencyLevels.get(cfName);
037          return wcf != null ? wcf : defaultWriteConsistencyLevel;
038        }
039      }
040    
041      public void setReadCfConsistencyLevels(Map<String, HConsistencyLevel> columnFamilyConsistencyLevels) {
042        this.readCfConsistencyLevels = columnFamilyConsistencyLevels;
043      }
044    
045      public void setWriteCfConsistencyLevels(Map<String, HConsistencyLevel> columnFamilyConsistencyLevels) {
046        this.writeCfConsistencyLevels = columnFamilyConsistencyLevels;
047      }
048    
049      public void setConsistencyLevelForCfOperation(HConsistencyLevel consistencyLevel,
050          String columnFamily,
051          OperationType operationType) {
052        if ( operationType.equals(OperationType.READ)) {
053          readCfConsistencyLevels.put(columnFamily, consistencyLevel);
054        } else {
055          writeCfConsistencyLevels.put(columnFamily, consistencyLevel);
056        }
057        log.info("{} ConsistencyLevel set to {} for ColumnFamily {}",
058            new Object[]{operationType.toString(),consistencyLevel.toString(),columnFamily});
059      }
060    
061      public void setDefaultReadConsistencyLevel(HConsistencyLevel defaultReadConsistencyLevel) {
062        this.defaultReadConsistencyLevel = defaultReadConsistencyLevel;
063      }
064    
065      public void setDefaultWriteConsistencyLevel(HConsistencyLevel defaultWriteConsistencyLevel) {
066        this.defaultWriteConsistencyLevel = defaultWriteConsistencyLevel;
067      }
068    
069    
070    
071    }
