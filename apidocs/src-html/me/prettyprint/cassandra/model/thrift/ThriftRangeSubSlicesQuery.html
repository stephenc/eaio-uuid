024    /**
025     * A query for the thrift call get_range_slices for subcolumns of supercolumns
026     *
027     * @author Ran Tavory
028     *
029     */
030    public final class ThriftRangeSubSlicesQuery<K,SN,N,V> extends AbstractSliceQuery<K,N,V,OrderedRows<K,N,V>>
031      implements RangeSubSlicesQuery<K, SN, N, V>{
032    
033      private final Serializer<SN> sNameSerializer;
034      private final HKeyRange<K> keyRange;
035      private SN superColumn;
036    
037      public ThriftRangeSubSlicesQuery(Keyspace k, Serializer<K> keySerializer, Serializer<SN> sNameSerializer,
038          Serializer<N> nameSerializer, Serializer<V> valueSerializer) {
039        super(k, keySerializer, nameSerializer, valueSerializer);
040        Assert.notNull(sNameSerializer, "sNameSerializer cannot be null");
041        this.sNameSerializer = sNameSerializer;
042        keyRange = new HKeyRange<K>(keySerializer);
043      }
044    
045    
046      @Override
047      public RangeSubSlicesQuery<K,SN,N,V> setKeys(K start, K end) {
048        keyRange.setKeys(start, end);
049        return this;
050      }
051    
052      @Override
053      public RangeSubSlicesQuery<K,SN,N,V> setRowCount(int rowCount) {
054        keyRange.setRowCount(rowCount);
055        return this;
056      }
057    
058      @Override
059      public  RangeSubSlicesQuery<K,SN,N,V> setSuperColumn(SN sc) {
060        Assert.notNull(sc, "sc can't be null");
061        superColumn = sc;
062        return this;
063      }
064    
065      @Override
066      public QueryResult<OrderedRows<K,N, V>> execute() {
067        Assert.notNull(columnFamilyName, "columnFamilyName can't be null");
068        Assert.notNull(superColumn, "superColumn cannot be null");
069    
070        return new QueryResultImpl<OrderedRows<K,N,V>>(keyspace.doExecute(
071            new KeyspaceOperationCallback<OrderedRows<K,N,V>>() {
072              @Override
073              public OrderedRows<K,N,V> doInKeyspace(KeyspaceService ks) throws HectorException {
074                ColumnParent columnParent = new ColumnParent(columnFamilyName);
075                columnParent.setSuper_column(sNameSerializer.toByteBuffer(superColumn));
076                Map<K, List<Column>> thriftRet = keySerializer.fromBytesMap(
077                    ks.getRangeSlices(columnParent, getPredicate(), keyRange.toThrift()));
078                return new OrderedRowsImpl<K,N,V>((LinkedHashMap<K, List<Column>>) thriftRet, columnNameSerializer, valueSerializer);
079              }
080            }), this);
081      }
082    
083      @Override
084      public String toString() {
085        return "RangeSuperSlicesQuery(" + keyRange + super.toStringInternal() + ")";
086      }
087    
088      @SuppressWarnings("unchecked")
089      @Override
090      public RangeSubSlicesQuery<K, SN, N, V> setColumnNames(N... columnNames) {
091        return (RangeSubSlicesQuery<K, SN, N, V>) super.setColumnNames(columnNames);
092      }
093    
094      @SuppressWarnings("unchecked")
095      @Override
096      public RangeSubSlicesQuery<K, SN, N, V> setRange(N start, N finish, boolean reversed, int count) {
097        return (RangeSubSlicesQuery<K, SN, N, V>) super.setRange(start, finish, reversed, count);
098      }
099    
100      @SuppressWarnings("unchecked")
101      @Override
102      public RangeSubSlicesQuery<K, SN, N, V> setColumnFamily(String cf) {
103        return (RangeSubSlicesQuery<K, SN, N, V>) super.setColumnFamily(cf);
104      }
105    
106    }
