34  /**
35   * Test case for the {@link com.eaio.uuid.MACAddressParser} class.
36   *
37   * @author <a href="mailto:jb@eaio.com">Johann Burkard</a>
38   * @version $Id: MACAddressParserTest.java,v 1.4 2008/02/20 07:37:40 Johann Exp $
39   */
40  public class MACAddressParserTest extends TestCase {
41  
42      public static void main(String[] args) {
43          junit.textui.TestRunner.run(MACAddressParserTest.class);
44      }
45  
46      public void testParse() {
47          assertNull(MACAddressParser.parse("kalifornia blorb blob"));
48          assertNull(MACAddressParser.parse("bla: keks: blorb: blubbix:a"));
49          assertEquals(
50                  "0:3:ba:1b:c4:74",
51                  MACAddressParser.parse("blablorb (10.20.30.40) at 0:3:ba:1b:c4:74 permanent published"));
52          assertEquals(
53                  "00:12:F0:21:F1:57",
54                  MACAddressParser.parse("        Physikalische Adresse . . . . . . : 00-12-F0-21-F1-57"));
55      }
56  
57      public void testParse2() {
58          String addr;
59  
60          addr = "blablorb (10.20.30.40) at 0:3:ba:1b:c4:74 permanent published";
61          for (int i = addr.length(); i > 40; --i) {
62              assertEquals("0:3:ba:1b:c4:74",
63                      MACAddressParser.parse(addr.substring(0, i)));
64          }
65          assertEquals("0:3:ba:1b:c4:7", MACAddressParser.parse(addr.substring(0,
66                  40)));
67          for (int i = 39; i != 0; --i) {
68              assertNull(MACAddressParser.parse(addr.substring(0, i)));
69          }
70      }
71  
72      public void testParseLanscan() {
73          assertNull(MACAddressParser.parse("Hardware Station        Crd  Hdw   Net-Interface    NM   MAC       HP-DLPI DLPI"));
74          assertNull(MACAddressParser.parse("Path     Address        In#  State NamePPA          ID   Type      Support Mjr#"));
75          assertEquals(
76                  "001560045000",
77                  MACAddressParser.parse("0/1/2/0  0x001560045000 0    UP    lan0 snap0       1    ETHER       Yes   119"));
78      }
79  
80      /**
81       * <code>Description . . . . . . . . . . . : Broadcom 440x 10/100 Integrated Controller</code>
82       * 
83       * Mail from Graham Matthews, 2008-01-29.
84       *
85       */
86      public void testBroadcom() {
87          assertNull(MACAddressParser.parse("Description . . . . . . . . . . . : Broadcom 440x 10/100 Integrated Controller"));
88          assertEquals(
89                  "00:1C:23:AD:D1:5A",
90                  MACAddressParser.parse("Physical Address. . . . . . . . . : 00-1C-23-AD-D1-5A"));
91      }
92  
93      public void testMacOSXLeopard() throws IOException {
94          ByteArrayOutputStream bos = new ByteArrayOutputStream();
95          InputStream is = null;
96          try {
97              is = getClass().getResourceAsStream("/macosx-leopard.txt");
98              byte[] buf = new byte[1024];
99              int len;
100             while ((len = is.read(buf)) != -1) {
101                 bos.write(buf, 0, len);
102             }
103         }
104         finally {
105             if (is != null) {
106             try {
107                 is.close();
108             }
109             catch (IOException ex) {}
110             }
111         }
112         byte[] buf = bos.toByteArray();
113 
114         String macosxLeopard = new String(buf, 0);
115         String[] lines = macosxLeopard.split("\n");
116         String mac = null;
117         for (String line : lines) {
118             mac = MACAddressParser.parse(line);
119             if (mac != null) {
120                 break;
121             }
122         }
123         assertEquals("00:1e:c2:00:43:51", mac);
124     }
