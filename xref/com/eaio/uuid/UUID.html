39  /**
40   * Creates UUIDs according to the DCE Universal Token Identifier specification.
41   * <p>
42   * All you need to know:
43   * <pre>
44   * UUID u = new UUID();
45   * </pre>
46   *
47   * @see <a href="http://www.opengroup.org/onlinepubs/9629399/apdxa.htm">

109     /**
110      * Parses a textual representation of a UUID.
111      * <p>
112      * No validation is performed. If the {@link CharSequence} is shorter than 36 characters,
113      * {@link ArrayIndexOutOfBoundsException}s will be thrown.
114      *
115      * @param s the {@link CharSequence}, may not be <code>null</code>
116      */
117     public UUID(CharSequence s) {
118         this(Hex.parseLong(s.subSequence(0, 18)), Hex.parseLong(s.subSequence(
119                 19, 36)));
120     }

122     /**
123      * Compares this UUID to another Object. Throws a {@link ClassCastException} if
124      * the other Object is not an instance of the UUID class. Returns a value
125      * smaller than zero if the other UUID is "larger" than this UUID and a value
126      * larger than zero if the other UUID is "smaller" than this UUID.
127      *
128      * @param t the other UUID, may not be <code>null</code>
129      * @return a value &lt; 0, 0 or a value &gt; 0
130      * @see java.lang.Comparable#compareTo(java.lang.Object)
131      * @throws ClassCastException
132      */
133     public int compareTo(UUID t) {
134         if (this == t) {
135             return 0;
136         }
137         if (time > t.time) {
138             return 1;
139         }
140         if (time < t.time) {
141             return -1;
142         }
143         if (clockSeqAndNode > t.clockSeqAndNode) {
144             return 1;
145         }
146         if (clockSeqAndNode < t.clockSeqAndNode) {
147             return -1;
148         }
149         return 0;
150     }

186     /**
187      * Appends a String representation of this to the given {@link StringBuffer} or
188      * creates a new one if none is given.
189      *
190      * @param in the StringBuffer to append to, may be <code>null</code>
191      * @return a StringBuffer, never <code>null</code>
192      * @see #toAppendable(Appendable)
193      */
194     public StringBuffer toStringBuffer(StringBuffer in) {
195         StringBuffer out = in;
196         if (out == null) {
197             out = new StringBuffer(36);
198         }
199         else {
200             out.ensureCapacity(out.length() + 36);
201         }
202         return (StringBuffer) toAppendable(out);
203     }

205     /**
206      * Appends a String representation of this object to the given {@link Appendable} object.
207      * <p>
208      * For reasons I'll probably never understand, Sun has decided to have a number of I/O classes implement
209      * Appendable which forced them to destroy an otherwise nice and simple interface with {@link IOException}s.
210      * <p>
211      * I decided to ignore any possible IOExceptions in this method.
212      *
213      * @param a the Appendable object, may be <code>null</code>
214      * @return an Appendable object, defaults to a {@link StringBuilder} if <code>a</code> is <code>null</code>
215      */
216     public Appendable toAppendable(Appendable a) {
217         Appendable out = a;
218         if (out == null) {
219             out = new StringBuilder(36);
220         }
221         try {
222             Hex.append(out, (int) (time >> 32)).append('-');
223             Hex.append(out, (short) (time >> 16)).append('-');
224             Hex.append(out, (short) time).append('-');
225             Hex.append(out, (short) (clockSeqAndNode >> 48)).append('-');
226             Hex.append(out, clockSeqAndNode, 12);
227         }
228         catch (IOException ex) {
229             // What were they thinking?
230         }
231         return out;
232     }

234     /**
235      * Returns a hash code of this UUID. The hash code is calculated by XOR'ing the
236      * upper 32 bits of the time and clockSeqAndNode fields and the lower 32 bits of
237      * the time and clockSeqAndNode fields.
238      *
239      * @return an <code>int</code> representing the hash code
240      * @see java.lang.Object#hashCode()
241      */
242     @Override
243     public int hashCode() {
244         return (int) ((time >> 32) ^ time ^ (clockSeqAndNode >> 32) ^ clockSeqAndNode);
245     }
